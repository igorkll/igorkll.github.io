the terminal allows you to organize input and output from your computer in text form
contains the input history that you can scroll through using the arrows
it makes a horn sound when displaying the "BELL" symbol (7)
component name - terminal

methods:

  • terminal.clear() - clears the terminal
  • terminal.read():string|nil - returns the user's last input if it is present, otherwise it returns nil
  • terminal.write(string) - outputs a line to the terminal
  • terminal.setSyntax(boolean) - enables syntax highlighting in the terminal input field (default: false)
  • terminal.isSyntax():boolean - returns the value set using terminal.setSyntax (default: false)
            local lineChar = string.char(13)
local bellChar = string.char(7)

local terminal = getComponent("terminal")
terminal.read()
terminal.clear()
terminal.write("#ffff00terminal demo code" .. lineChar)

function callback_loop()
    local text = terminal.read()
    if text then
        if text == "/beep" then
            terminal.write(string.char(7))
        elseif text == "/clear" then
            terminal.clear()
        end
        terminal.write("#00ff00> " .. text .. bellChar .. lineChar)
    end

    if _endtick then
        terminal.clear()
    end
end
